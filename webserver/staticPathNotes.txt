Setup Static Web Server Using Path Module

-- using Path (node.js module_
-- Path provides utilities for working with file and directory paths
-- usage: 	- const path = require('path') <--sets the variable for 'path'
			- app.use(express.static(path.join(__dirname, 'public'))); <-- sets the static path to find all of the pages we will create



-- eliminates the need for routes to be placed in the app.js:

		app.get('/', function(req, res){
		res.send('This is the home page');
	});

	//this is exactly the same, but uses ES6 syntax
	app.get('/about', (req, res) => {
		res.send('<h1>This page injects html directly into the request</h1>');
	});

	app.get('/users/:name', (req, res) => {
		let user = req.params.name;
		res.send('<h1> This page takes in a route parameter. Here, the parameter is ' +user +'</h1>');
	});
